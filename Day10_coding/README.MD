```java
    public static boolean isPerfectSquare(int N) {
        if (N < 1) return false;

        int left = 1, right = N;
        while (left <= right) {
            int mid = left + (right - left) / 2;

            if (mid * mid == N) {
                return true;
            } else if (mid * mid < N) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }

        return false;
    }

    public static void main(String[] args) {
        int N = 25;
        System.out.println(isPerfectSquare(N));
    }
}
```
